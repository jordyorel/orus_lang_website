Fibonacci Sequence

                    
// Calculate Fibonacci numbers
function fibonacci(n) {
    if (n <= 1) return n;
    return fibonacci(n-1) + fibonacci(n-2);
}

for (let i = 0; i < 10; i++) {
    print(fibonacci(i));
}
                    

                    
                

                
                

                    
Sorting Algorithm

                    
// Bubble sort implementation
function bubbleSort(arr) {
    let n = arr.length;
    for (let i = 0; i < n; i++) {
        for (let j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                // Swap elements
                let temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
    return arr;
}

let numbers = [64, 34, 25, 12, 22, 11, 90];
print("Original array: " + numbers);
print("Sorted array: " + bubbleSort(numbers));

Getting Started with Orus

                        
Orus is a modern programming language designed to be expressive, safe, and easy to learn. This guide will help you get started with Orus programming.

Syntax Basics

                        
Orus has a clean, readable syntax inspired by modern programming languages.

                        
Comments

                        
// This is a single-line comment

/* This is a
   multi-line comment */
                        

                        
Variables

                        
let name = "John";  // Variable declaration
const PI = 3.14159; // Constant declaration
                        

                    

                    
                    

                        
Data Types

                        
Orus supports various data types for different kinds of values.

                        
Basic Types

                        
let num = 42;        // Integer
let pi = 3.14159;    // Float
let name = "Alice";  // String
let isActive = true; // Boolean
let nothing = null;  // Null value
                        

                        
Collections

                        
let numbers = [1, 2, 3, 4, 5];  // Array
let person = {               // Object
    name: "Bob",
    age: 30,
    isStudent: false
};
                        

                    

                    
                    

                        
Functions

                        
Functions in Orus are first-class citizens and can be defined in multiple ways.

                        
Function Declaration

                        
function greet(name) {
    return "Hello, " + name + "!";
}

print(greet("World")); // Outputs: Hello, World!
                        

                        
Anonymous Functions

                        
let add = function(a, b) {
    return a + b;
};

print(add(5, 3)); // Outputs: 8
                        

                    

                    
                    

                        
Control Flow

                        
Orus provides standard control flow statements for conditional execution and loops.

                        
Conditionals

                        
let age = 20;

if (age >= 18) {
    print("Adult");
} else if (age >= 13) {
    print("Teenager");
} else {
    print("Child");
}
                        

                        
Loops

                        
// For loop
for (let i = 0; i < 5; i++) {
    print(i);
}

// While loop
let count = 0;
while (count < 5) {
    print(count);
    count++;
}
